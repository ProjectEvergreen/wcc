static parseAttribute = value => value.charAt(0) === '{' || value.charAt(0) === '['
  ? JSON.parse(value)
  : value !== '' && !isNaN(+value)
    ? parseInt(value, 10)
    : value === 'true' || value === 'false'
      ? value === 'true' ? true : false
      : value;
attributeChangedCallback(name, oldValue, newValue) {
  this[name].set(Counter.parseAttribute(newValue));
}
